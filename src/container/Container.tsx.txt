import {
  defineComponent,
} from 'vue-demi';
import { createNamespace } from '../utils';
import { NavBar as VanNavBar } from 'vant';


const [name, bem] = createNamespace('container');


export default defineComponent({
  name: name,
  props: {
    title: {
      type: String,
      default: 'title',
    },
    background: {
      type: [String],
    },
  },
  components: {
    VanNavBar
  },
  setup(props: any, { emit, slots }: any) {
    const clickLeft = () => {
      emit('clickLeft');
    };
    const slotObj = {
      right: slots.headRight
    }
    return () => (
      <div class="vp-container">
        <VanNavBar
          title={props.title}
          left-arrow
          onClick-left={clickLeft}
        >
          {slotObj}
        </VanNavBar>
        <div class="vp-container__content " style={{ background: props.background }}>
          {slots.default?.()}
        </div>
      </div >
    );
  },
});
